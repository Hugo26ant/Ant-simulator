function calcular() { const notasCuaderno = procesarNotas('cuaderno'); const notasParciales = procesarNotas('parciales'); const notasTrimestral = procesarNotas('trimestral'); // 1. MEDIA del Cuaderno (vale el 10% del total) const mediaCuaderno = notasCuaderno.length > 0 ? (notasCuaderno.reduce((a, b) => a + b, 0) / notasCuaderno.length) : 0; const puntosCuaderno = mediaCuaderno * 0.10; // 2. MEDIA de los Parciales (vale el 15% del total) const mediaParciales = notasParciales.length > 0 ? (notasParciales.reduce((a, b) => a + b, 0) / notasParciales.length) : 0; const puntosParciales = mediaParciales * 0.15; // 3. Trimestral (vale el 50% del total) const notaTrimestral = notasTrimestral.length > 0 ? notasTrimestral[0] : 0; const puntosTrimestral = notaTrimestral * 0.50; // 4. NOTA FINAL // Nota: Sumamos los puntos. Si falta un 25% (porque 10+15+50=75), // el sistema actual asume que ese 25% son puntos que no se han ganado. const notaFinal = puntosCuaderno + puntosParciales + puntosTrimestral; const divRes = document.getElementById('resultado'); divRes.style.display = 'block'; divRes.innerHTML = `Nota Final: ${notaFinal.toFixed(2)}`; divRes.className = notaFinal >= 5 ? 'aprobado' : 'suspenso'; }
